// SCENE: TitleScene

// -----------------------------------------------------------------------------

class TitleScene extends Phaser.Scene {

  constructor() {

    super({ key: 'TitleScene' });

  }


  preload() {

    // Load images for both characters

    this.load.image('blaire', 'assets/daughter.png'); 

    this.load.image('drew', 'assets/son.png'); 

    // Load background music

    this.load.audio('bgMusic', 'assets/bgMusic.mp3');

  }


  create() {

    // Play background music (loop)

    this.bgMusic = this.sound.add('bgMusic', { loop: true, volume: 0.5 });

    this.bgMusic.play();


    // Title text

    this.add.text(200, 100, "Monkeys World", {

      fontSize: '48px',

      fill: '#000',

      fontStyle: 'bold'

    });


    // Character selection instructions

    this.add.text(120, 250, "Choose Your Character:", {

      fontSize: '32px',

      fill: '#000'

    });

    this.add.text(140, 320, "Press 1 for Blaire", {

      fontSize: '24px',

      fill: '#000'

    });

    this.add.text(140, 360, "Press 2 for Drew", {

      fontSize: '24px',

      fill: '#000'

    });

    this.add.text(20, 20, "TitleScene loaded!", {

      fontSize: '16px',

      fill: '#fff'

    });


    // Keyboard input for character selection

    this.input.keyboard.on('keydown-ONE', () => {

      this.startGame('blaire');

    });

    this.input.keyboard.on('keydown-TWO', () => {

      this.startGame('drew');

    });


    // If someone taps/clicks the screen, default to Blaire

    this.input.on('pointerdown', () => {

      this.startGame('blaire');

    });

  }


  startGame(characterKey) {

    // Pass the chosen character to Level1

    this.scene.start('Level1', { character: characterKey, level: 1, score: 0 });

  }

}


// -----------------------------------------------------------------------------

// SCENE: Level1

// -----------------------------------------------------------------------------

class Level1 extends Phaser.Scene {

  constructor() {

    super({ key: 'Level1' });

  }


  init(data) {

    this.characterKey = data.character || 'blaire'; 

    this.level = data.level || 1;

    this.score = data.score || 0;

  }


  create() {

    // Game dimensions

    this.gameWidth = this.sys.game.config.width;

    this.gameHeight = this.sys.game.config.height;


    // Create player sprite based on chosen character

    this.player = this.physics.add.sprite(100, 450, this.characterKey);

    this.player.setCollideWorldBounds(true);

    this.player.setScale(0.5);


    // Ground

    this.ground = this.add.rectangle(

      this.gameWidth / 2,

      this.gameHeight - 20,

      this.gameWidth,

      40,

      0x654321

    );

    this.physics.add.existing(this.ground, true);

    this.physics.add.collider(this.player, this.ground);


    // Obstacles group

    this.obstacles = this.physics.add.group();

    this.physics.add.collider(this.obstacles, this.ground);

    this.physics.add.overlap(this.player, this.obstacles, this.hitObstacle, null, this);


    // Collectibles group (for 🤟 emoji)

    this.collectibles = this.physics.add.group();

    this.physics.add.collider(this.collectibles, this.ground);

    this.physics.add.overlap(this.player, this.collectibles, this.collectItem, null, this);


    // Score text

    this.scoreText = this.add.text(16, 16, 'Score: ' + this.score, {

      fontSize: '32px',

      fill: '#000'

    });


    // Timer

    this.levelDuration = 120; // 2 minutes

    this.timerText = this.add.text(650, 16, 'Time: ' + this.levelDuration, {

      fontSize: '32px',

      fill: '#000'

    });

    this.time.addEvent({

      delay: 1000,

      callback: this.updateTimer,

      callbackScope: this,

      loop: true

    });


    // Input for jump

    this.input.on('pointerdown', () => this.jump());

    this.spaceKey = this.input.keyboard.addKey(Phaser.Input.Keyboard.KeyCodes.SPACE);

    this.spaceKey.on('down', () => this.jump());


    // Spawn obstacles every 1.5 seconds

    this.time.addEvent({

      delay: 1500,

      callback: this.addObstacle,

      callbackScope: this,

      loop: true

    });


    // Spawn collectibles (🤟) every 3 seconds

    this.time.addEvent({

      delay: 3000,

      callback: this.addCollectible,

      callbackScope: this,

      loop: true

    });


    // Debug text

    this.add.text(20, 60, "Level1 loaded!", {

      fontSize: '16px',

      fill: '#fff'

    });

  }


  update() {

    // Auto-run

    this.player.setVelocityX(200);

  }


  jump() {

    if (this.player.body.touching.down) {

      this.player.setVelocityY(-400);

    }

  }


  addObstacle() {

    // Randomly choose between hockey net (🥅) or lacrosse stick (🥍)

    const obstacleEmoji = Math.random() < 0.5 ? "🥅" : "🥍";

    const obstacle = this.add.text(this.gameWidth, 520, obstacleEmoji, {

      fontSize: '64px'

    });

    this.physics.add.existing(obstacle);

    obstacle.body.setVelocityX(-250);

    obstacle.body.setAllowGravity(false);

    this.obstacles.add(obstacle);

  }


  addCollectible() {

    // Spawn the "I love you" sign language emoji 🤟

    const collectible = this.add.text(this.gameWidth, 480, '🤟', {

      fontSize: '48px'

    });

    this.physics.add.existing(collectible);

    collectible.body.setVelocityX(-200);

    collectible.body.setAllowGravity(false);

    this.collectibles.add(collectible);

  }


  hitObstacle(player, obstacle) {

    // If the player is above the obstacle => successful jump

    if (player.y < obstacle.y) {

      this.score += 10;

      this.scoreText.setText('Score: ' + this.score);

      obstacle.destroy();

    } else {

      // Game over

      this.scene.start('GameOverScene', { score: this.score });

    }

  }


  collectItem(player, collectible) {

    // Collect 🤟 => +20 points

    this.score += 20;

    this.scoreText.setText('Score: ' + this.score);

    collectible.destroy();

  }


  updateTimer() {

    this.levelDuration--;

    this.timerText.setText('Time: ' + this.levelDuration);

    if (this.levelDuration <= 0) {

      // Go to Level2

      this.scene.start('Level2', {

        character: this.characterKey,

        level: 2,

        score: this.score

      });

    }

  }

}

// SCENE: GameOverScene

class GameOverScene extends Phaser.Scene {

  constructor() {

    super({ key: 'GameOverScene' });

  }


  init(data) {

    this.finalScore = data.score || 0;

    this.victory = data.victory || false;

  }


  create() {

    const message = this.victory ? "You completed all levels!" : "Game Over!";


    this.add.text(200, 200, message, {

      fontSize: '48px',

      fill: '#000',

      fontStyle: 'bold'

    });


    this.add.text(200, 300, `Final Score: ${this.finalScore}`, {

      fontSize: '32px',

      fill: '#000'

    });


    this.add.text(150, 400, "Tap or Press SPACE to Restart", {

      fontSize: '24px',

      fill: '#000'

    });


    this.input.keyboard.on('keydown-SPACE', () => {

      this.restartGame();

    });

    this.input.on('pointerdown', () => {

      this.restartGame();

    });

  }


  restartGame() {

    // Return to TitleScene

    this.scene.start('TitleScene');

  }

}
